31inline auto bindNullStatementParameter(soci::values& values, const db::StatementParameter& parameter) -> void
32{
33 switch (parameter.nullType)
34 {
35 case model::ColumnType::Bool:
36 case model::ColumnType::Char:
37 case model::ColumnType::UnsignedChar:
38 case model::ColumnType::Short:
39 case model::ColumnType::UnsignedShort:
40 case model::ColumnType::Int:
41 bindTypedNullStatementParameter(values, parameter.name, int{});
42 return;
43 case model::ColumnType::UnsignedInt:
44 case model::ColumnType::UnsignedLongLong:
45 bindTypedNullStatementParameter(values, parameter.name, static_cast<unsigned long long>(0));
46 return;
47 case model::ColumnType::LongLong:
48 bindTypedNullStatementParameter(values, parameter.name, static_cast<long long>(0));
49 return;
50 case model::ColumnType::Float:
51 case model::ColumnType::Double:
52 bindTypedNullStatementParameter(values, parameter.name, 0.0);
53 return;
54 case model::ColumnType::String:
55 bindTypedNullStatementParameter(values, parameter.name, std::string{});
56 return;
57 case model::ColumnType::Uuid:
58 case model::ColumnType::Unknown:
59 case model::ColumnType::OneToOne:
60 throw std::invalid_argument{"Cannot bind NULL parameter with unsupported column type"};
61 }
62
63 throw std::invalid_argument{"Cannot bind NULL parameter with unsupported column type"};
64}

66inline auto bindStatementParameter(soci::values& values, const db::StatementParameter& parameter) -> void
